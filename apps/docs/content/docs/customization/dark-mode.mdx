Theme management is flexible and personal aspect of application design, so we donâ€™t include a theme provider by default. We recommend using [`next-themes`](https://www.npmjs.com/package/next-themes) to easily add dark mode or custom theming to your Next.js app.

<Steps>
  <Step>
  ### Install next-themes

  To get started with dark mode, install the `next-themes` package:

  ```package-install
  next-themes
  ```
  </Step>
  <Step>
  ### Create a Theme Provider

  Create a `ThemeProvider` component to manage the theme state.
  
  ```json doc-gen:file
  {
    "file": "./examples/theme/theme-provider.tsx",
    "codeblock": {
      "lang": "tsx",
      "meta": "title=\"components/theme-provider.tsx\""
    }
  }
  ```
  </Step>
  <Step>
  ### Wrap your root layout

  Wrap your root layout with the `ThemeProvider` to enable theme switching.

  ```tsx
  import { ThemeProvider } from "@/components/theme-provider" // [!code highlight]

  export default function RootLayout({ children }: RootLayoutProps) {
    return (
        <html lang="en" suppressHydrationWarning>
          <body>
            <ThemeProvider // [!code highlight]
              attribute="class" // [!code highlight]
              defaultTheme="system" // [!code highlight]
              enableSystem
              disableTransitionOnChange
            >
              {children}
            </ThemeProvider>
          </body>
        </html>
    )
  }
  ```
  
  </Step>
  <Step>
  ### Add a Theme Toggle Button

  Add a button to toggle between light and dark modes.
