## Usage

```tsx
import { CalendarDatePicker } from "@mijn-ui/react"
```

```tsx
const [selectedDateRange, setSelectedDateRange] = useState({
  from: new Date(new Date().getFullYear(), 0, 1),
  to: new Date(),
})

<CalendarDatePicker
  color="default"
  date={selectedDateRange}
  onDateSelect={setSelectedDateRange}
/>
```
